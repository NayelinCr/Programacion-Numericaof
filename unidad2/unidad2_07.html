# Datos
x <- c(2010, 2012, 2014, 2016)

# Funcion para diferencias divididas
diferencias_divididas <- function(x, y) {
  n <- length(x)
  tabla <- matrix(0, n, n)
  tabla[,1] <- y
  for (j in 2:n) {
    for (i in 1:(n-j+1)) {
      tabla[i, j] <- (tabla[i+1, j-1] - tabla[i, j-1]) / (x[i+j-1] - x[i])
    }
  }
  return(tabla)
}

# Evaluacion en x=2017
newton_interp <- function(x, y, valor) {
  tabla <- diferencias_divididas(x, y)
  n <- length(x)
  resultado <- tabla[1, 1]
  producto <- 1
  for (i in 1:(n-1)) {
    producto <- producto * (valor - x[i])
    resultado <- resultado + tabla[1, i+1] * producto
  }
  return(resultado)
}

poblacion_2017 <- newton_interp(x, y, 2017)
poblacion_2017
RESULTADO: P(2017) ‚âà 106.9 mil habitantes

# Ejercicio 1 Hermite cubico entre dos puntos
hermite_cubico_2p <- function(x0, y0, dy0, x1, y1, dy1, xq) {
  h <- x1 - x0
  t <- (xq - x0)/h
  h00 <- 2*t^3 - 3*t^2 + 1
  h10 <- t^3 - 2*t^2 + t
  h01 <- -2*t^3 + 3*t^2
  h11 <- t^3 - t^2
  Hx <- h00*y0 + h10*h*dy0 + h01*y1 + h11*h*dy1
  return(Hx)
}

# Hermite general (n puntos con derivadas)
hermite_general_prompt <- function(x, y, dy, xq) {
  n <- length(x)
  m <- 2*n
  z <- rep(x, each=2)
  Q <- matrix(0, nrow=m, ncol=m)
  
  # Primera columna
  Q[,1] <- rep(y, each=2)
  
  # Segunda columna
  for(i in 1:m) {
    if(i %% 2 == 0) {
       Q[i, 2] <- dy[i/2] # Derivada en fila par
    } else {
       if(i > 1) {
          Q[i, 2] <- (Q[i,1] - Q[i-1,1]) / (z[i] - z[i-1])
       }
    }
  }
  # Ajuste manual primera derivada
  Q[2, 2] <- dy[1] 

  # Resto de la tabla
  for(j in 3:m) {
    for(i in j:m) {
      Q[i, j] <- (Q[i, j-1] - Q[i-1, j-1]) / (z[i] - z[i-j+1])
    }
  }
  
  coeffs <- diag(Q) # Coeficientes en la diagonal
  
  # Evaluacion
  res <- coeffs[m]
  for(k in (m-1):1) {
    res <- res * (xq - z[k]) + coeffs[k]
  }
  return(list(valor=res))
}

x <- c(1.0, 2.0, 3.0)
y <- c(2.0, 3.0, 5.0)
dy <- c(0.5, 1.0, 1.5)
xq <- 2.5

res_std <- hermite_general_prompt(x, y, dy, xq)
print(round(res_std$valor, 7))
RESULTADO: f(2.5) ‚âà 3.9375

Comparativa de M√©todos [cite: 1942]

M√©todo Ventajas Desventajas
Lineal Sencillo y r√°pido. Ideal para intervalos cortos. Poca precisi√≥n en curvas. Error alto fuera de rango.
Lagrange Polinomio √∫nico y suave. No requiere sistemas de ecuaciones. Ineficiente para recalcular si se agregan puntos (Fen√≥meno de Runge).
Newton Eficiente computacionalmente. F√°cil actualizaci√≥n con nuevos datos. Pierde precisi√≥n con muchos puntos. Sensible a datos dispersos.
Hermite Curva muy suave y realista al usar derivadas. Requiere conocer las derivadas. Complejo de implementar.
