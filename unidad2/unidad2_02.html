# C√ìDIGO R: TRIPLE GR√ÅFICO (f(x), f'(x), Pr√≥ximo x) VS. ITERACI√ìN

#1. INSTALACI√ìN Y CARGA DE LIBRER√çAS
if(!require(ggplot2)) install.packages("ggplot2")
if(!require(reshape2)) install.packages("reshape2")
library(ggplot2)
library(reshape2)

#2. PAR√ÅMETROS DEL M√âTODO
alpha <- 0.01 
x <- 3 # **REVISA ESTAS L√çNEAS MANUALMENTE!**
max_iter <- 50

# Funci√≥n y derivada
f <- function(x) x^2
f_prime <- function(x) 2 * x

#3. C√ÅLCULOS ITERATIVOS (Generando su tabla)
# Se incluye una iteraci√≥n extra para el punto inicial (i=0)
resultados <- data.frame(
    Iteracion = 0:max_iter,
    x = numeric(max_iter + 1),
    f_x = numeric(max_iter + 1),
    fprima_x = numeric(max_iter + 1),
    gradiente_f_x = numeric(max_iter + 1)
)

# Inicializar la primera fila (i=0)
resultados$x[1] <- x
resultados$f_x[1] <- f(x)
resultados$fprima_x[1] <- f_prime(x)
resultados$gradiente_f_x[1] <- x - alpha * f_prime(x)

# Bucle para las iteraciones restantes
for (i in 1:max_iter) {
    # El valor de 'x' para la nueva iteraci√≥n se toma de la columna 'gradiente_f_x' anterior
    x_anterior <- resultados$gradiente_f_x[i]

    # Guardar los valores en la fila actual
    resultados$x[i + 1] <- x_anterior
    resultados$f_x[i + 1] <- f(x_anterior)
    resultados$fprima_x[i + 1] <- f_prime(x_anterior)

    # Calcular el nuevo valor de 'gradiente_f_x' (Pr√≥ximo x)
    resultados$gradiente_f_x[i + 1] <- x_anterior - alpha * f_prime(x_anterior)
}

# La √∫ltima fila de 'gradiente_f_x' no se usa
resultados$gradiente_f_x[max_iter + 1] <- NA

#4. PREPARACI√ìN Y GR√ÅFICO DEL RESULTADO

# A. Reestructurar el data frame para ggplot
datos_plot <- melt(resultados,
    id.vars = "Iteracion",
    measure.vars = c("f_x", "fprima_x", "gradiente_f_x"),
    variable.name = "Mtrica",
    value.name = "Valor"
)

# B. Renombrar las etiquetas
datos_plot$Mtrica <- factor(datos_plot$Mtrica,
    levels = c("f_x", "fprima_x", "gradiente_f_x"),
    labels = c("f(x)=x^2",
               "f'(x) - Gradiente",
               "Pr√≥ximo Valor de x (Columna E)")
)

# C. Generar el gr√°fico con ggplot2
ggplot(datos_plot, aes(x = Iteracion, y = Valor, color = Mtrica)) +
    geom_line(size = 1) +
    geom_point(size = 2, alpha = 0.7) +
    labs(
        title = "Evoluci√≥n de f(x), f'(x) y el Pr√≥ximo x a lo largo de las Iteraciones",
        subtitle = "Simulaci√≥n del Gradiente Descendente (Œ±=0.01) de su hoja de c√°lculo",
        x = "Iteraci√≥n (i)",
        y = "Valor de la M√©trica"
    ) +
    theme_minimal() +

                                  "f'(x) - Gradiente" = "red",

2. C√≥digo para Dos Funciones: Gradiente Descendente

# C√ìDIGO R: GRADIENTE DESCENDENTE COMPARATIVO (Dos Funciones)

#1. INSTALACI√ìN Y CARGA DE LIBRER√çAS
if(!require(ggplot2)) install.packages("ggplot2")
if (!require(reshape2)) install.packages("reshape2")
library(ggplot2)
library(reshape2)

#2. PAR√ÅMETROS GLOBALES
alpha <- 0.01 # Tasa de aprendizaje
x0 <- 3 # Valor inicial
max_iter <- 50 # N√∫mero de iteraciones

#3. DEFINICI√ìN DE FUNCIONES

# Primera funci√≥n
f1 <- function(x) x^2
f1_prime <- function(x) 2 * x

# Segunda funci√≥n
f2 <- function(x) (x - 2)^2 + 1
f2_prime <- function(x) 2 * (x - 2)

#4. FUNCI√ìN PARA CALCULAR EL GRADIENTE DESCENDENTE
gradiente_descendente <- function(f, f_prime, x0, alpha, max_iter, nombre_fun) {
    resultados <- data.frame(
        Iteracion = 0:max_iter,
        x = numeric(max_iter + 1),
        f_x = numeric(max_iter + 1),
        fprima_x = numeric(max_iter + 1),
        gradiente_f_x = numeric(max_iter + 1),
        Funcion = nombre_fun
    )

    # Inicializaci√≥n
    x <- x0
    resultados$x[1] <- x
    resultados$f_x[1] <- f(x)
    resultados$fprima_x[1] <- f_prime(x)
    resultados$gradiente_f_x[1] <- x - alpha * f_prime(x)

    # Iteraciones
    for (i in 1:max_iter) {
        x_anterior <- resultados$gradiente_f_x[i]
        resultados$x[i + 1] <- x_anterior
        resultados$f_x[i + 1] <- f(x_anterior)
        resultados$fprima_x[i + 1] <- f_prime(x_anterior)
        resultados$gradiente_f_x[i + 1] <- x_anterior - alpha * f_prime(x_anterior)
    }
    resultados$gradiente_f_x[max_iter + 1] <- NA
    return(resultados)
}

#5. EJECUCI√ìN PARA AMBAS FUNCIONES
resultados_f1 <- gradiente_descendente(f1, f1_prime, x0, alpha, max_iter, "f(x)=x^2")
resultados_f2 <- gradiente_descendente(f2, f2_prime, x0, alpha, max_iter, "f(x)=(x-2)^2 + 1")

# Unir resultados
resultados_totales <- rbind(resultados_f1, resultados_f2)

#6. REESTRUCTURAR DATOS PARA ggplot
datos_plot <- melt(resultados_totales,
    id.vars = c("Iteracion", "Funcion"),
    measure.vars = c("f_x", "fprima_x", "gradiente_f_x"),
    variable.name = "Mtrica",
    value.name = "Valor"
)

# Etiquetas m√°s legibles
datos_plot$Mtrica <- factor(datos_plot$Mtrica,
    levels = c("f_x", "fprima_x", "gradiente_f_x"),
    labels = c("f(x)", "f'(x)", "Pr√≥ximo x")
)

#7. GRAFICAR LOS RESULTADOS
ggplot(datos_plot, aes(x = Iteracion, y = Valor, color = Mtrica, linetype = Funcion)) +
    geom_point(size = 1.5, alpha = 0.6) +
    geom_line(size = 1) +
    labs(
        title = "Comparaci√≥n del Gradiente Descendente para Dos Funciones",
        subtitle = sprintf("Œ±=%.3f, x0=%.1f, Iteraciones =%d", alpha, x0, max_iter),
        x = "Iteraci√≥n (i)",
        y = "Valor de la M√©trica"
    ) +
    scale_color_manual(values = c("f(x)" = "blue", "f'(x)" = "red", "Pr√≥ximo x" = "darkgreen")) +
    theme_minimal(base_size = 13) +
    theme(
        legend.position = "top",
        plot.title = element_text(face = "bold"),
        panel.grid.minor = element_blank()
    )
